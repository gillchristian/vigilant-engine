const strToDate = (str: string): Date => {
  const [, y, m, d] = /([0-9]{4})([0-9]{2})([0-9]{2})/.exec(str)

  return new Date(parseInt(y, 10), parseInt(m, 10) - 1, parseInt(d, 10) + 1)
}
